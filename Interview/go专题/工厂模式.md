# 基于Go多态实现的工厂模式

基于Go的interface实现的多态

```Go
    type animal interface {
    	sleep() string
    }
    type dog struct{}
    type pig struct{}
    func (d *dog) sleep() string {
    	return "dog sleep"
    }
    func (p *pig) sleep() string {
    	return "pig sleep"
    }

```

返回方法

```Go
    func Factory(a string) animal {
    	switch a {
    	case "dog":
    		return &dog{}
    	case "pig":
    		return &pig{}
    	default:
    		// fmt.Println("not have this animal")
    		panic("no such animal")
    	}
    }
```
